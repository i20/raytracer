COLOR

    Color + Color : Color
    Color - Color : Color

    Color * Color : Color
    Color * float : Color

MATRIX

    Matrix * Matrix : Matrix
    Matrix * Vector : Vector
    Matrix * Point : Point
    Matrix * Ray : Ray

POINT

    Point - Point : Vector
    Point + Vector : Point

VECTOR

    Vector ^ Vector : Vector (vectorial product)
    Vector * Vector : float (scalar product)
    Vector + Vector : Vector
    Vector - Vector : Vector

    Vector * float : Vector
    Vector / float : Vector
